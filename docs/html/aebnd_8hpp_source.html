403 template <typename T, typename PMT, typename PVT>
404 inline typename AEBND<T,PMT,PVT>::STATUS
405 AEBND<T,PMT,PVT>::setup
406 ( std::ostream&os )
407 {
408  _issetup = false;
409 
410  // Check problem size
411  _nvar = _var.size();
412  _ndep = _dep.size();
413  if( !_ndep || _sys.size()!=_ndep ) return FAILURE;
414 
415  // (Re)size variable arrays
416  const unsigned nVAR = _nvar+_ndep;
417  if( _pVAR.size() < nVAR ){
418  _pVAR.resize(nVAR);
419  _IVAR.clear(); _IREF.clear();
420  _PMVAR.clear(); _PMREF.clear();
421  }
422 
423  // Populate variable arrays
424  unsigned ivar=0;
425  for( unsigned idep=0; idep<_ndep; idep++, ivar++ ){
426  _pVAR[ivar] = _dep[idep];
427 #ifdef MC__AEBND_DEBUG
428  os << "dependent #" << idep << ": " << _pVAR[ivar] << std::endl;
429 #endif
430  }
431  for( unsigned ipar=0; ipar<_nvar; ipar++, ivar++ ){
432  _pVAR[ivar] = _var[ipar];
433 #ifdef MC__AEBND_DEBUG
434  os << "independent #" << ipar << ": " << _pVAR[ivar] << std::endl;
435 #endif
436  }
437 
438  // Initialize DAG operations w/ block decomposition
439  _pAE.resize(_ndep);
440  for( unsigned i=0; i<_ndep; i++ ) _pAE[i] = _sys[i];
441  _blockdec( os );
442  if( _singstruct ) return SINGULAR;
443 
444  _opAE.resize(_bAE.size());
445  std::vector<const FFVar*>::iterator it = _pAEJAC.begin();
446  for( ; it != _pAEJAC.end(); ++it ) delete[] *it;
447  _pAEJAC.clear(); _opAEJAC.clear();
448  _islinblk.resize(_bAE.size());
449  _bandblk.resize(_bAE.size());
450 
451  _maxpAE = _maxopAE = 0;
452  for( unsigned iblk=0; iblk<_bAE.size(); iblk++ ){
453  // Initialize block operations and Jacobian
454 #ifdef MC__AEBND_DEBUG
455  double blkoper_cpu = -time();
456 #endif
457  const unsigned ndepblk = ( iblk==_bAE.size()-1? _ndep: _bAE[iblk+1] ) - _bAE[iblk];
458  const unsigned posblk = _ndep - _bAE[iblk] - ndepblk;
459  _opAE[iblk] = _dag->subgraph( ndepblk, _pAE.data()+posblk );
460 
461  const FFVar* pJACi = _dag->FAD( ndepblk, _pAE.data()+posblk, ndepblk, _pVAR.data()+posblk );
462  _pAEJAC.push_back( pJACi );
463  std::list<const FFOp*> opJACi = _dag->subgraph( ndepblk*ndepblk, pJACi );
464  _opAEJAC.push_back( opJACi );
465 
466  if( _maxpAE < ndepblk ) _maxpAE = ndepblk;
467  if( _maxopAE < _opAE[iblk].size() ) _maxopAE = _opAE[iblk].size();
468  if( _maxopAE < opJACi.size() ) _maxopAE = opJACi.size();
469 #ifdef MC__AEBND_DEBUG
470  blkoper_cpu += time();
471 #endif
472 
473  // Detect block linearity
474 #ifdef MC__AEBND_DEBUG
475  double blkprop_cpu = -time();
476 #endif
477  std::vector<FFDep> depblk(ndepblk), varblk(_pVAR.size()-posblk);
478  for( unsigned i(0); i<ndepblk; i++ ) varblk[i].indep(i);
479  _dag->eval( _opAE[iblk], ndepblk, _pAE.data()+posblk, depblk.data(),
480  _pVAR.size()-posblk, _pVAR.data()+posblk, varblk.data() );
481  _islinblk[iblk] = true;
482  _bandblk[iblk].first = _bandblk[iblk].second = 0;
483  for( unsigned i=0; i<ndepblk; i++ ){
484  std::map<int,bool>::const_iterator cit = depblk[i].dep().begin();
485  for( ; cit != depblk[i].dep().end(); ++cit ){
486  if( !(*cit).second ) _islinblk[iblk] = false; // Detecting linearity
487  if( _bandblk[iblk].first < (int)i-(*cit).first ) // Updating lower band width
488  _bandblk[iblk].first = (int)i-(*cit).first;
489  if( _bandblk[iblk].second < (*cit).first-(int)i ) // updating upper band width
490  _bandblk[iblk].second = (*cit).first-(int)i;
491  }
492  }
493 #ifdef MC__AEBND_DEBUG
494  blkprop_cpu += time();
495  std::cout << "Block #" << iblk << ": "
496  << (_islinblk[iblk]?"linear":"nonlinear") << ", bandwidth "
497  << _bandblk[iblk].first << "," << _bandblk[iblk].second << std::endl
498  << "CPU: " << blkoper_cpu << " (oper) " << blkprop_cpu << " (prop)\n";
499 #endif
500  }
501 
502  _maxopAE*=2;
503  _issetup = true;
504  return NORMAL;
505 }

507 template <typename T, typename PMT, typename PVT>
508 inline void
509 AEBND<T,PMT,PVT>::_blockdec
510 ( std::ostream&os )
511 {
512  // Compute structure with mc::FFDep
513  const unsigned nVAR = _pVAR.size();
514  std::vector<FFDep> VAR(nVAR), SYS(_ndep);
515  for( unsigned i(0); i<_ndep; i++ ) VAR[i].indep(i);
516  _dag->eval( _ndep, _pAE.data(), SYS.data(), nVAR, _pVAR.data(), VAR.data() );
517 
518  // Detect linearity and populate sparse arrays
519  std::vector<int> IP(_ndep), LENR(_ndep), ICN;
520  _islindep = true;
521  for( unsigned i=0; i<_ndep; i++ ){
522  IP[i] = ICN.size()+1;
523  LENR[i] = SYS[i].dep().size();
524  std::map<int,bool>::const_iterator cit = SYS[i].dep().begin();
525  for( ; cit != SYS[i].dep().end(); ++cit ){
526  ICN.push_back( (*cit).first+1 );
527  if( !(*cit).second ) _islindep = false; // Detecting linearity
528  }
529  }
530 #ifdef MC__AEBND_DEBUG
531  std::cout << "Linearity: " << (_islindep?'Y':'N') << std::endl;
532 #endif
533 
534  // Make a row permutation to remove nonzeros on diagonal: MC21A
535  int N = IP.size(), LICN = ICN.size();
536  std::vector<int> IPERM(_ndep), IW(4*_ndep);
537 #ifdef MC__AEBND_DISABLE_MC21A
538  for( unsigned i=0; i<_ndep; i++ ) IPERM[i] = i+1;
539  _singstruct = false;
540 #else
541  int NUMNZ;
542  mc21a_( &N, ICN.data(), &LICN, IP.data(), LENR.data(), IPERM.data(),
543  &NUMNZ, IW.data() );
544  _singstruct = NUMNZ<N? true: false;
545 #ifdef MC__AEBND_DEBUG
546  std::cout << "Structural singularity: " << (_singstruct?'Y':'N') << std::endl;
547 #endif
548  // return if structurally singular matrix or block decomposition not requested
549  if( _singstruct || !options.BLKDEC ){
550  _bAE.resize(1); _bAE[0] = 0;
551  _bVAR.resize(_ndep); _bVARrev.resize(_ndep);
552  for( unsigned i=0; i<_ndep; i++ ) _bVAR[i] = _bVARrev[i] = i;
553  std::vector<FFVar> pAEblk(_ndep);
554  for( unsigned i=0; i<_ndep; i++ ) pAEblk[i] = _pAE[IPERM[i]-1];
555  _pAE.swap(pAEblk);
556 
557  // Display permuted system structure
558  if( options.DISPLAY >= 2 ){
559  os << std::endl << "Number of Blocks: 1" << std::endl;
560  os << " ";
561  for( unsigned j=0; j<_ndep; j++ )
562  os << " " << std::setw(3) << j;
563  os << std::endl;
564  for( unsigned i=0; i<_ndep; i++ ){
565  os << std::setw(3) << IPERM[i]-1;
566  for( unsigned j=0; j<_ndep; j++ )
567  os << std::setw(3) << " " << (SYS[IPERM[i]-1].dep(j).first?"X":" ");
568  os << std::endl;
569  }
570  }
571 
572  return;
573  }
574 
575  // Permute order of equation system using IPERM (!!Fortran style indices!!)
576  ICN.clear();
577  for( unsigned i=0; i<_ndep; i++ ){
578  IP[i] = ICN.size()+1;
579  LENR[i] = SYS[IPERM[i]-1].dep().size();
580  std::map<int,bool>::const_iterator cit = SYS[IPERM[i]-1].dep().begin();
581  for( ; cit != SYS[IPERM[i]-1].dep().end(); ++cit )
582  ICN.push_back( (*cit).first+1 );
583  }
584 #ifdef MC__AEBND_DEBUG
585  std::cout << "Row reordering: ";
586  for( unsigned i=0; i<_ndep; i++) std::cout << " " << IPERM[i];
587  std::cout << std::endl;
588 #endif
589 #endif
590 
591  // Make a block lower-triangular decomposition: MC13D
592  int NUM;
593  std::vector<int> IOR(_ndep), IB(_ndep);
594  _bAE.resize(_ndep);
595  mc13d_( &N, ICN.data(), &LICN, IP.data(), LENR.data(), IOR.data(),
596  IB.data(), &NUM, IW.data() );
597  _bAE.resize(NUM);
598  for( int i=0; i<NUM; i++ ) _bAE[i] = IB[i]-1;
599 #ifdef MC__AEBND_DEBUG
600  std::cout << "Number of blocks in permuted matrix: " << NUM << std::endl;
601  std::cout << "Row/Column reordering: ";
602  for( unsigned i=0; i<_ndep; i++) std::cout << " " << IOR[i];
603  std::cout << std::endl;
604 #endif
605 
606  // Permute order of equation system AND variables using IOR
607  // and keep track of inverse permutation
608  std::vector<FFVar> pAEblk(_ndep), pVARblk(_pVAR);
609  _bVAR.resize(_ndep); _bVARrev.resize(_ndep);
610  for( unsigned i=0; i<_ndep; i++ ){
611  pAEblk[i] = _pAE[IPERM[IOR[_ndep-i-1]-1]-1];
612  pVARblk[i] = _pVAR[IOR[_ndep-i-1]-1];
613  _bVAR[IOR[i]-1] = _ndep-i-1;
614  _bVARrev[_ndep-i-1] = IOR[i]-1;
615  }
616  //for( int k=0; k<NUM; k++ ){
617  // const unsigned iL = IB[0]-1, iU = ( k==NUM-1? _ndep: IB[k+1]-1 );
618  // for( unsigned i=iL; i<iU; i++ ){
619  // pVARblk[i] = _pVAR[IOR[_ndep-iU+(i-iL)]-1];
620  // _bVAR[IOR[i]-1] = _ndep-iU+(i-iL);
621  // }
622  //}
623  _pAE.swap(pAEblk);
624  _pVAR.swap(pVARblk);
625 
626  // Display permuted system structure
627  if( options.DISPLAY >= 2 ){
628  std::cout << std::endl << "Number of Blocks: " << NUM << std::endl;
629  os << " ";
630  for( unsigned j=0; j<_ndep; j++ )
631  os << " " << std::setw(3) << IOR[_ndep-j-1]-1;
632  os << std::endl;
633  for( unsigned i=0; i<_ndep; i++ ){
634  os << std::setw(3) << IPERM[IOR[_ndep-i-1]-1]-1;
635  for( unsigned j=0; j<_ndep; j++ )
636  os << std::setw(3) << " "
637  << (SYS[IPERM[IOR[_ndep-i-1]-1]-1].dep(IOR[_ndep-j-1]-1).first?"X":" ");
638  os << std::endl;
639  }
640  }

677 {
678  if( !_issetup ) return false;
679

682  if( !_IVAR.size() ) _IVAR.resize(nVAR);
683  if( !_PMVAR.size() ) _PMVAR.resize(nVAR);
684  if( !_PMREF.size() ) _PMREF.resize(nVAR);

693

934 inline typename AEBND<T,PMT,PVT>::STATUS
935 AEBND<T,PMT,PVT>::_gs
936 ( const unsigned iblk, const unsigned ndepblk, U*var, U*opf, U*f, U*ref,
937  V*opjacf, V*jacf, V*jacvar, bool usekraw, std::ostream&os )
938 {
939  const unsigned posblk = _ndep - _bAE[iblk] - ndepblk;
940  U*varblk = var + posblk;
941  U*refblk = ref + posblk;
942 
943 #ifdef MC__AEBND_DEBUG
944  std::cout << "Initial bounds:\n";
945  for( unsigned i=0; i<ndepblk; i++ ) std::cout << varblk[i] << std::endl;
946 #endif
947 
948  std::vector<U> G(ndepblk*ndepblk), b(ndepblk), varblk0;
949  for( unsigned iter=0; iter<options.MAXIT; iter++ ){
950  if( _stats_ae.maxIter <= iter ) _stats_ae.maxIter = iter+1;
951 
952  // Update reference
953  _reference( ndepblk, posblk, var, ref, jacvar );
954 
955  // (Re)compute preconditionned LHS matrix (only if needed) and RHS vector
956  _precondlin( iblk, ndepblk, posblk, G, b, opf, f, ref,
957  opjacf, (!iter||!_islinblk[iblk]?jacf:0), jacvar, os );
958 
959  // Keep track of current iterate
960  varblk0.assign( varblk, varblk+ndepblk );
961 
962  try{
963  // Apply componentwise Krawczyk step
964  if( usekraw ){
965  for( unsigned i=0; i<ndepblk; i++ ){
966  U Xk, temp(0.);
967  for( unsigned j=0; j<ndepblk; j++ ){
968  if( j != i ) temp -= G[_ndx(i,j,ndepblk)] * ( varblk[j] - refblk[j] );
969  else temp += ( 1. - G[_ndx(i,i,ndepblk)] ) * ( varblk[j] - refblk[j] );
970  }
971  Xk = refblk[i] + b[i] + temp;
972  if( !Op<U>::inter( varblk[i], Xk, varblk[i] ) ) return EMPTY;
973  }
974  }
975 
976  // Apply componentwise Gauss-Seidel step
977  else{
978  for( unsigned i=0; i<ndepblk; i++ ){
979  U Xk, temp(0.);
980  if( Op<U>::l(G[_ndx(i,i,ndepblk)]) > 0.
981  || Op<U>::u(G[_ndx(i,i,ndepblk)]) < 0. ){
982  for( unsigned j=0; j<ndepblk; j++ )
983  if( j != i ) temp += G[_ndx(i,j,ndepblk)] * ( varblk[j] - refblk[j] );
984  Xk = refblk[i] + ( b[i] - temp ) / G[_ndx(i,i,ndepblk)];
985  }
986  else{
987  for( unsigned j=0; j<ndepblk; j++ ){
988  if( j != i ) temp -= G[_ndx(i,j,ndepblk)] * ( varblk[j] - refblk[j] );
989  else temp += ( 1. - G[_ndx(i,i,ndepblk)] ) * ( varblk[j] - refblk[j] );
990  }
991  Xk = refblk[i] + b[i] + temp;
992  }
993  if( !Op<U>::inter( varblk[i], Xk, varblk[i] ) ) return EMPTY;
994  //else if( options.DISPLAY >= 2 )
995  // os << " Skipped Iteration #" << iter+1
996  // << " for Variable X" << _bVARrev[posblk+i] << std::endl;
997  }
998  }
999 #ifdef MC__AEBND_DEBUG
1000  std::cout << "Iter #" << iter << " bounds:\n";
1001  for( unsigned i=0; i<ndepblk; i++ ) std::cout << varblk[i] << std::endl;
1002 #endif
1003 
1004  // Display
1005  if( options.DISPLAY >= 2 ){
1006  os << std::endl << "Block #" << iblk+1 << " Iteration #" << iter+1 << ":\n";
1007  for( unsigned i=0; i<ndepblk; i++ )
1008  //os << " X" << _bVAR[_ndep-posblk-i-1] << ": " << varblk[i] << std::endl;
1009  os << " X" << _bVARrev[posblk+i] << ": " << varblk[i] << std::endl;
1010  }
1011 
1012  // Check convergence
1013  if( _cvgtest( ndepblk, varblk, varblk0.data() ) ) break;
1014 
1015  }
1016  catch(...){ return FAILURE; }
1017  }
1018 
1019  return NORMAL;

1037 template <typename U>
1038 inline typename AEBND<T,PMT,PVT>::STATUS
1039 AEBND<T,PMT,PVT>::_ge
1040 ( const unsigned iblk, const unsigned ndepblk, U*var, U*op, U*f, U*jacf,
1041  std::ostream&os )
1042 {
1043  if( !_stats_ae.maxIter ) _stats_ae.maxIter = 1;
1044  const unsigned posblk = _ndep - _bAE[iblk] - ndepblk;
1045  U*varblk = var + posblk;
1046 
1047  // Compute preconditionned LHS matrix and RHS vector
1048  std::vector<U> G(ndepblk*ndepblk), b(ndepblk);
1049  for( unsigned i=0; i<ndepblk; i++ ) varblk[i] = 0.;
1050  _precondlin( iblk, ndepblk, posblk, G, b, op, f, var,
1051  op, jacf, var, os );
1052 
1053  try{
1054  // Perform forward substitution
1055  for( unsigned k=0; k<ndepblk-1; k++ ){
1056  if( Op<U>::l( G[_ndx(k,k,ndepblk)] ) <= 0.
1057  && Op<U>::u( G[_ndx(k,k,ndepblk)] ) >= 0. ) return SINGULAR;
1058 
1059  for( unsigned i=k+1; i<ndepblk; i++ ){
1060  U factor = G[_ndx(i,k,ndepblk)] / G[_ndx(k,k,ndepblk)];
1061  for( unsigned j=k+1; j<ndepblk; j++ )
1062  G[_ndx(i,j,ndepblk)] -= factor * G[_ndx(k,j,ndepblk)];
1063  b[i] -= factor * b[k];
1064  }
1065  }
1066

1068  if( Op<U>::l( G[_ndx(ndepblk-1,ndepblk-1,ndepblk)] ) <= 0.
1069  && Op<U>::u( G[_ndx(ndepblk-1,ndepblk-1,ndepblk)] ) >= 0. )
1070  return SINGULAR;
1071  varblk[ndepblk-1] = b[ndepblk-1] / G[_ndx(ndepblk-1,ndepblk-1,ndepblk)];
1072  for( unsigned i=ndepblk; i>0; i-- ){
1073  U sum = b[i-1];
1074  for( unsigned j=i; j<ndepblk; j++ )
1075  sum -= G[_ndx(i-1,j,ndepblk)] * varblk[j];
1076  varblk[i-1] = sum / G[_ndx(i-1,i-1,ndepblk)];
1077  }
1078 #ifdef MC__AEBND_DEBUG
1079  std::cout << "GE bounds:\n";
1080  for( unsigned i=0; i<ndepblk; i++ ) std::cout << varblk[i] << std::endl;
1081 #endif
1082  }
1083  catch(...){ return FAILURE; }
1084  return NORMAL;

1102 {

1108  STATUS flag = NORMAL;
1109  for( unsigned iblk=0; iblk<_bAE.size() && flag == NORMAL; iblk++ ){

1121  flag = _ge( iblk, ndepblk, _IVAR.data(), _Iop.data(), _If.data(), _Idfdx.data(), os );
1122  }
1123 
1124  // A priori box supplied and linear case

1158

1161  // Check initial box is given
1162  if( !Ix0 ) throw Exceptions( Exceptions::APRIORI );

1170  // Check initial box is given
1171  if( !Ix0 ) throw Exceptions( Exceptions::APRIORI );
1172  for( unsigned i=0; i<ndepblk; i++ ) _IVAR[posblk+i] = Ix0[_bVARrev[posblk+i]];
