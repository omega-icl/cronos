333 template <typename T>
334 inline double
335 NLCP_GUROBI<T>::solve
336 ( const T*P0, std::ostream&os )
337 {
338  // Keep track of execution times
339  stats.reset();
340 
341  // Set variable types and exclusions from B&B
342  NLGO_GUROBI<T>::options = options;
343  _set_SetInvoptions();
344  std::vector<CVar<T>> CVP( P0, P0+_nvar );
345  SetInv<CVar<T>>::variables( _nvar, CVP.data(), _CMexcl );
346 
347  // Set-up NLP relaxed solver
348  _tvar.clear();
349  std::vector<T> P( P0, P0+_nvar );
350  NLGO_GUROBI<T>::_set_polrelax( P.data(), 0, true );
351 #if defined (MC__NLCP_DEBUG)
352  std::cout << *_dag;
353  std::cout << NLGO_GUROBI<T>::_POLenv;
354  //return 0;
355 #endif
356 
357  // Set-up reduced-space Chebyshev model (if applicable)
358  _nrvar = BASE_NLP::_var.size();
359  _nrdep = BASE_NLP::_dep.size();
360  if( _nrdep && options.CMREDORD ){
361  if( _CMrenv && (_CMrenv->nvar() != _nrvar || _CMrenv->nord() != options.CMREDORD) ){
362  delete _CMrenv; _CMrenv = 0;
363  _CMrbas.clear();
364  }
365  if( !_CMrenv ){
366  // Create reduce-space Chebyshev model and populate with [-1,1] scaled variables
367  _CMrenv = new CModel<T>( _nrvar, options.CMREDORD );
368  for( unsigned i=0; i<_nrvar; i++ )
369  _CMrbas.push_back( CVar<T>( _CMrenv, i, T(-1,1) ) );
370  }
371  _CMrvar.resize( _nrvar );
372  _CMrdep.resize( _nrdep );
373  AEBND<T,CModel<T>,CVar<T> >::options = options.CMREDOPT;
374  AEBND<T,CModel<T>,CVar<T> >::setup();
375  }
376 
377  // Run set-inversion algorithm
378  double res = SetInv<CVar<T>>::solve( os );
379  return res;
380 }

383 inline typename SetInv<CVar<T>>::STATUS
384 NLCP_GUROBI<T>::assess
385 ( SetInvNode<CVar<T>>*node )
386 {
387 #if defined (MC__NLCP_DEBUG)
388  std::cout << "\nInitial Box:\n";
389  for( unsigned i=0; i<_nvar; i++ )
390  if( _exclude_vars.find(i) == _exclude_vars.end() )
391  std::cout << node->P()[i] << std::endl;
392 #endif
393 
394  // Strong branching case
395  unsigned DOMREDMAX = options.DOMREDMAX;
396  if( node->strongbranch() && options.STGBCHDRMAX
397  && options.STGBCHDRMAX < options.DOMREDMAX ){
398  options.DOMREDMAX = options.STGBCHDRMAX;
399 #if defined (MC__NLCP_DEBUG)
400  std::cout << "on-going strong branching\n";
401 #endif
402  }
403 
404  // Node assessment and domain contraction
405  std::vector<T> IP;
406  for( auto it=node->P().begin(); it!=node->P().end(); ++it )
407  IP.push_back( (*it).B() );
408  auto status = _subproblem_assess( IP.data() );
409  node->P().assign( IP.begin(), IP.end() ); // Could be buggy!!!
410 #if defined (MC__NLCP_DEBUG)
411  std::cout << "\nReduced Box:\n";
412  for( unsigned i=0; i<_nvar; i++ )
413  if( _exclude_vars.find(i) == _exclude_vars.end() )
414  std::cout << node->P()[i] << std::endl;
415 #endif
416 
417  // Strong branching case
418  if( node->strongbranch() ){
419  options.DOMREDMAX = DOMREDMAX;
420  return status;
421  }
422 
423  // Subspace reduction
424  if( status != SetInv<CVar<T>>::UNDETERMINED || !_nrdep || !options.CMREDORD )
425  return status;
426  _subproblem_project( node->P().data(), node->depend() );
427 #if defined (MC__NLCP_DEBUG)
428  std::cout << "\nProjected Box:\n";
429  for( unsigned i=0; i<_nvar; i++ )
430  if( _exclude_vars.find(i) == _exclude_vars.end() )
431  std::cout << node->P()[i] << std::endl;
432 #endif
433 
434  return status;
435 }

438 inline typename SetInv<CVar<T>>::STATUS
439 NLCP_GUROBI<T>::_subproblem_assess
440 ( T*P )
441 {
442  // Apply domain contraction
443  // Do NOT test for infeasibility here, because contraction problem may
444  // become infeasible due to round-off in LP solver
445  unsigned *tvar = !_tvar.empty()? _tvar.data(): 0;
446  unsigned nred = 0;
447  auto status = _assess( P, nred, tvar, &options.CTRBACKOFF, false, true );
448  if( status != SetInv<CVar<T>>::UNDETERMINED ) return status;
449 
450  // Solve relaxed feasibility problem
451  //NLGO_GUROBI<T>::_update_polrelax( P, tvar, true );
452  //std::cout << "Solving relaxed feasibility problem:\n"; {int dum; std::cin >> dum; }
453  NLGO_GUROBI<T>::relax( P, tvar, 0, false, true );
454  switch( NLGO_GUROBI<T>::get_status() ){
455  // objective contains minimum backoffs
456 #ifdef MC__USE_CPLEX
457  case IloAlgorithm::Optimal:
458 #else
459  case GRB_OPTIMAL:
460 #endif
461  //std::cout << "Feasibility:" << NLGO_GUROBI<T>::get_objective() << std::endl;
462  if( NLGO_GUROBI<T>::get_objective() > options.LPOPTIMTOL )
463  return SetInv<CVar<T>>::OUTER;
464  break; // need more test to find out if INNER or UNDETERMINED
465  // infeasibility may not happen w/ backoffs
466 #ifdef MC__USE_CPLEX
467  case IloAlgorithm::Infeasible:
468 #else
469  case GRB_INFEASIBLE:
470 #endif
471  default:
472  return SetInv<CVar<T>>::FAILURE;
473  }
474  return SetInv<CVar<T>>::UNDETERMINED;
475 }

477 template <typename T> inline typename SetInv<CVar<T>>::STATUS
478 NLCP_GUROBI<T>::_assess
479 ( T*P, unsigned&nred, const unsigned*tvar, const double*inc,
480  const bool reset, const bool feastest )
481 {
482  // Main loop for relaxation and domain reduction
483  std::vector<T> P0;
484  for( nred=0; nred < options.DOMREDMAX; nred++ ){
485  // Update relaxation
486  if( reset ) NLGO_GUROBI<T>::_set_polrelax( P, tvar, feastest );
487  else NLGO_GUROBI<T>::_update_polrelax( P, tvar, feastest );
488  // Perform inclusion test
489  //std::cout << "Solving inclusion test:\n"; {int dum; std::cin >> dum; }
490  auto status = _inclusion_test( _CMctr );
491  if( status != SetInv<CVar<T>>::UNDETERMINED ) return status;
492  // Perform domain reduction
493  P0.assign( P, P+_nvar );
494  NLGO_GUROBI<T>::_contract( P, tvar, inc, false, feastest );
495  if( !NLGO_GUROBI<T>::optimal_status() ) return status;
496  double vred = NLGO_GUROBI<T>::_reducrel( _nvar, P, P0.data() );
497  if( vred < options.DOMREDTHRES ) break;
498 #ifdef MC__NLGO_GUROBI_DEBUG
499  std::cout << "Reduction #" << nred << ": " << vred*1e2 << "%\n";
500 #endif
501  }
502  return SetInv<CVar<T>>::UNDETERMINED;
503 }

505 template <typename T>
506 template <typename U>
507 inline typename SetInv<CVar<T>>::STATUS
508 NLCP_GUROBI<T>::_inclusion_test
509 ( const std::vector<U>&C ) const
510 {
511  if( C.empty() ) return SetInv<CVar<T>>::UNDETERMINED;
512 
513  // Function enclosure and inclusion tests
514  bool flag1 = true, flag2 = true;
515  auto itc = std::get<0>(NLGO_GUROBI<T>::_ctr).begin();
516  for( unsigned j=0; flag2 && itc!=std::get<0>(NLGO_GUROBI<T>::_ctr).end(); ++itc, j++ ){
517  switch( (*itc) ){
518  case EQ: if( Op<U>::u(C[j]) < 0. )
519  flag2 = false;
520  else if( Op<U>::l(C[j]) < 0. )
521  flag1 =false;
522  // no break;
523  case LE: if( Op<U>::l(C[j]) > 0. )
524  flag2 = false;
525  else if( Op<U>::u(C[j]) > 0. )
526  flag1 =false;
527  break;
528  case GE: if( Op<U>::u(C[j]) < 0. )
529  flag2 = false;
530  else if( Op<U>::l(C[j]) < 0. )
531  flag1 =false;
532  break;
533  }
534  }
535  if (flag1 && flag2)
536  return SetInv<CVar<T>>::INNER;
537  if (!flag2)
538  return SetInv<CVar<T>>::OUTER;
539  return SetInv<CVar<T>>::UNDETERMINED;
