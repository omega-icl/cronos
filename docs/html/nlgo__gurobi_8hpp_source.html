787  _ctr = BASE_NLP::_ctr;

789  std::get<0>(_ctr).push_back( EQ );

878 {

919 template <typename T> inline int
920 NLGO_GUROBI<T>::_contract
921 ( T*P, const unsigned*tvar, const double*inc, const bool reset,
922  const bool feastest )
923 {
924  std::multimap<double,std::pair<unsigned,bool>> vardomred, vardomredupd;
925  std::pair<unsigned,bool> varini;
926  for( unsigned ip=0; ip<_nvar; ip++ ){
927  varini.first = ip;
928  varini.second = false; // lower bound
929  vardomred.insert( std::pair<double,std::pair<unsigned,bool>>(1.,varini) );
930  varini.second = true; // upper bound
931  vardomred.insert( std::pair<double,std::pair<unsigned,bool>>(1.,varini) );
932  }
933  if( reset ) _set_polrelax( P, tvar, feastest );
934 
935  // solve reduction subproblems from closest to farthest from bounds
936  unsigned nred=0;
937  for( ; !vardomred.empty(); nred++ ){
938  // upper/lower range reduction for current subproblem
939  auto itv = vardomred.begin();
940  const unsigned ip = (*itv).second.first;
941  const bool uplo = (*itv).second.second;
942  double pL = Op<T>::l( P[ip] ), pU = Op<T>::u( P[ip] );
943  contract( P, ip, uplo, tvar, inc, false, feastest );
944  if( !optimal_status() ) return get_status();
945  switch( uplo ){
946  case false: // lower bound
947  pL = get_objective();
948  if( options.DOMREDBKOFF > 0. ) pL -= options.DOMREDBKOFF;
949  if( !Op<T>::inter( P[ip], P[ip], T(pL,pU+1.) ) ) P[ip] = pU;
