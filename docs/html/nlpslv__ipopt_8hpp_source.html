294  bool setup()
295  {
296  // full set of decision variables (independent & dependent)
297  _var = BASE_NLP::_var;
298  _var.insert( _var.end(), _dep.begin(), _dep.end() );
299  _nvar = _var.size();
300 
301  // full set of constraints (main & equation system)
302  _ctr = BASE_NLP::_ctr;
303  for( auto its=_sys.begin(); its!=_sys.end(); ++its ){
304  std::get<0>(_ctr).push_back( EQ );
305  std::get<1>(_ctr).push_back( (*its) );
306  std::get<2>(_ctr).push_back( FFVar( _dag ) );
307  }
308  _nctr = std::get<0>(_ctr).size();
309 
310  // setup objective and constraint evaluation
311  _cleanup();
312  if( std::get<0>(_obj).size() != 1 ) throw Exceptions( Exceptions::NLPOBJ );
313  _op_f = _dag->subgraph( std::get<1>(_obj).size(), std::get<1>(_obj).data() );
314  _op_g = _dag->subgraph( std::get<1>(_ctr).size(), std::get<1>(_ctr).data() );
315 
316  // setup objective and constraint gradient evaluation
317  switch( options.GRADIENT ){
318  case Options::FORWARD: // Forward AD
319  _obj_grad = _dag->FAD( std::get<1>(_obj).size(), std::get<1>(_obj).data(),
320  _nvar, _var.data() );
321  _ctr_grad = _dag->SFAD( std::get<1>(_ctr).size(), std::get<1>(_ctr).data(),
322  _nvar, _var.data() );
323  break;
324  case Options::BACKWARD: // Backward AD
325  _obj_grad = _dag->BAD( std::get<1>(_obj).size(), std::get<1>(_obj).data(),
326  _nvar, _var.data() );
327  _ctr_grad = _dag->SBAD( std::get<1>(_ctr).size(), std::get<1>(_ctr).data(),
328  _nvar, _var.data() );
329  break;
330  }
331 #ifdef MC__NLPSLV_IPOPT_DEBUG
332  for( unsigned ie=0; ie<std::get<0>(_ctr_grad); ++ie )
333  std::cout << " dg[" << std::get<1>(_ctr_grad)[ie]
334  << "," << std::get<2>(_ctr_grad)[ie] << "]";
335  std::cout << std::endl;
336 #endif
337  _op_df = _dag->subgraph( _nvar, _obj_grad );
338  _op_dg = _dag->subgraph( std::get<0>(_ctr_grad), std::get<3>(_ctr_grad) );
339 
340  // setup Lagrangian Hessian evaluation
341  //if( options.HESSIAN == Options::EXACT ){
342  // form Lagrangian
343  FFVar lagr = std::get<1>(_obj)[0] * std::get<2>(_obj)[0];
344  for( unsigned ic=0; ic<_nctr; ic++ )
345  lagr += std::get<1>(_ctr)[ic] * std::get<2>(_ctr)[ic];
346  // differentiate twice
347  const FFVar* lagr_grad = _dag->BAD( 1, &lagr, _nvar, _var.data() );
348  _lagr_hess = _dag->SFAD( _nvar, lagr_grad, _nvar, _var.data(), true );
349  delete[] lagr_grad;
350 #ifdef MC__NLPSLV_IPOPT_DEBUG
351  _dag->output( _dag->subgraph( std::get<0>(_lagr_hess), std::get<3>(_lagr_hess) ), std::cout );
352  for( unsigned ie=0; ie<std::get<0>(_lagr_hess); ++ie )
353  std::cout << " d2L[" << std::get<1>(_lagr_hess)[ie]
354  << "," << std::get<2>(_lagr_hess)[ie] << "]: "
355  << std::get<3>(_lagr_hess)[ie];
356  std::cout << std::endl;
357 #endif
358  _op_d2L = _dag->subgraph( std::get<0>(_lagr_hess), std::get<3>(_lagr_hess) );
359  //}
360 
361  return true;
362  }

365  template <typename T>
366  Ipopt::ApplicationReturnStatus solve
367  ( const T*P, const double*p0=0 )
368  {
369  Ipopt::SmartPtr<Ipopt::IpoptApplication> IpoptApp
370  = new Ipopt::IpoptApplication();
371 
372  // Set (a few) IPOPT options
373  set_options( IpoptApp );
374 
375  // Keep track of bounds and initial guess
376  _data.set( _nvar, p0, P );
377 
378  // Run NLP solver
379  _solution.status = IpoptApp->Initialize();
380  if( _solution.status == Ipopt::Solve_Succeeded ){
381  _solution.status = IpoptApp->OptimizeTNLP( this );
382  }
383 
384  return _solution.status;
385  }

531 inline
532 bool
533 NLPSLV_IPOPT::get_bounds_info
534 ( Ipopt::Index n, Ipopt::Number* x_l, Ipopt::Number* x_u,
535  Ipopt::Index m, Ipopt::Number* g_l, Ipopt::Number* g_u )
536 {
537 #ifdef MC__NLPSLV_IPOPT_TRACE
538  std::cout << " NLPSLV_IPOPT::get_bounds_info\n";
539 #endif
540  // set variable bounds
541  unsigned ip = 0;
542  for( ; ip<_nvar; ip++ ){
543  x_l[ip] = ( _data.P? _data.P[ip].first: -INF );
544  x_u[ip] = ( _data.P? _data.P[ip].second: INF );
545 #ifdef MC__NLPSLV_IPOPT_DEBUG
546  std::cout << " x_l[" << ip << "] = " << x_l[ip]
547  << " x_u[" << ip << "] = " << x_u[ip] << std::endl;
548 #endif
549  }
550 
551  // set constraint bounds
552  auto itc = std::get<0>(_ctr).begin();
553  for( unsigned ic=0; itc!=std::get<0>(_ctr).end(); ++itc, ic++ ){
554  switch( (*itc) ){
555  case EQ: g_l[ic] = g_u[ic] = 0.; break;
556  case LE: g_l[ic] = -INF; g_u[ic] = 0.; break;
557  case GE: g_l[ic] = 0.; g_u[ic] = INF; break;
558  }
559 #ifdef MC__NLPSLV_IPOPT_DEBUG
560  std::cout << " g_l[" << ic << "] = " << g_l[ic]
561  << " g_u[" << ic << "] = " << g_u[ic] << std::endl;
562 #endif
563  }
564  return true;
565 }

742  nvar.push_back(n); pvar.push_back(_var.data()); vvar.push_back(x);

756  }
757  return true;
